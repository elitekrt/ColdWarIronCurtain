scripted_gui = {
	chem_weapon_button = {
		context_type = selected_state_context
		
		window_name = "chem_weapon_window"
		parent_window_token = selected_state_view
		
		ai_enabled = {
			always = no
		}
		visible = {
			always = yes
		}
		
		effects = {
			Chemical_Weapon_Button_click = {
				if = {
					limit = {
						ROOT = { has_country_flag = napalm_selected }
					}
					add_province_modifier = {
						static_modifiers = { napalm }
						province = {
							all_provinces = yes 
						}
					}

					add_to_variable = {
						napalm_total = 2
					}
					set_state_flag = napalm_attacked
					damage_building = {
						type = infrastructure
						damage = 1
					}
				}
				if = {
					limit = {
						ROOT = { has_country_flag = mustard_gas_selected }
					}
					add_province_modifier = {
						static_modifiers = { mustard_gas }
						province = {
							all_provinces = yes 
						}
					}
					add_to_variable = {
						mustard_gas_total = 2
					}
					set_state_flag = mustard_gas_attacked
				}	
				if = {
					limit = {
						ROOT = { has_country_flag = white_phosphorus_selected }
					}
					add_province_modifier = {
						static_modifiers = { white_phosphorus }
						province = {
							all_provinces = yes 
						}
					}
					add_to_variable = {
						white_phosphorus_total = 2
					}
					set_state_flag = white_phosphorus_attacked
				}	
				if = {
					limit = {
						ROOT = { has_country_flag = agent_orange_selected }
					}
					add_province_modifier = {
						static_modifiers = { agent_orange2 }
						province = {
							all_provinces = yes 
						}
					}
					add_to_variable = {
						agent_orange_total = 4
					}
					set_state_flag = agent_orange_attacked
				}	
				if = {
					limit = {
						ROOT = { has_country_flag = sarin_selected }
					}
					add_province_modifier = {
						static_modifiers = { sarin }
						province = {
							all_provinces = yes 
						}
					}
					add_to_variable = {
						sarin_total = 2
					}
					set_state_flag = sarin_attacked
				}	
				if = {
					limit = {
						ROOT = { has_country_flag = vx_selected }
					}

					if = {
						limit = { has_tech nbc_nerve_weapon_5 }
						add_province_modifier = {
							static_modifiers = { A_234 }
							province = {
								all_provinces = yes 
							}
						}
					}
					else_if = {
						limit = { has_tech nbc_nerve_weapon_4 }
						add_province_modifier = {
							static_modifiers = { GV }
							province = {
								all_provinces = yes 
							}
						}
					}
					else_if = {
						limit = { has_tech nbc_nerve_weapon_3 }
						add_province_modifier = {
							static_modifiers = { VM }
							province = {
								all_provinces = yes 
							}
						}
					}
					else_if = {
						limit = { has_tech nbc_nerve_weapon_2 }
						add_province_modifier = {
							static_modifiers = { EA_3990 }
							province = {
								all_provinces = yes 
							}
						}
					}
					else = {
						add_province_modifier = {
							static_modifiers = { vx }
							province = {
								all_provinces = yes 
							}
						}
					}
					add_to_variable = {
						vx_total = 12
					}
					set_state_flag = vx_attacked
				}	
			}
		}

		triggers = {
			Chemical_Weapon_Button_click_enabled = {
				check_variable = { ROOT.distance_check = 1 }
				is_in_array = {
					array = ROOT.WarCrimeTargetListChemical
					value = THIS.id
				}
				OR = {
					AND = {
						NOT = { has_state_flag = napalm_attacked }
						ROOT = { has_country_flag = napalm_selected }
					}
					AND = {
						NOT = { has_state_flag = mustard_gas_attacked }
						ROOT = { has_country_flag = mustard_gas_selected }
					}
					AND = {
						NOT = { has_state_flag = white_phosphorus_attacked }
						ROOT = { has_country_flag = white_phosphorus_selected }					
					}
					AND = {
						OR = {
							has_state_flag = boreal
							has_state_flag = tropical
						}
						NOT = { has_state_flag = agent_orange_attacked }
						ROOT = { has_country_flag = agent_orange_selected }					
					}
					AND = {
						NOT = { has_state_flag = sarin_attacked }
						ROOT = { has_country_flag = sarin_selected }					
					}
					AND = {
						NOT = { has_state_flag = vx_attacked }
						ROOT = { has_country_flag = vx_selected }					
					}					
				}
			}
		}
	}
}
